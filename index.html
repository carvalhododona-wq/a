\documentclass[conference]{IEEEtran} \usepackage{amsmath, amssymb, graphicx} \title{AURI-UCC: A Structural-Dynamic Framework for Regime Detection and Predictive Transition Analysis} \author{ \IEEEauthorblockN{Rogério Ezidio Carvalho Ferreira} \IEEEauthorblockA{ Verde Saúde Cosmic Technology\\ São José do Rio Preto, Brazil\\ Email: contato@verdesaude.org } } \begin{document} \maketitle \begin{abstract} This paper introduces AURI-UCC, a structural-dynamic modeling framework designed to detect and anticipate regime transitions in non-stationary systems. The model integrates semi-Markov persistence, structural derivatives based on ranking positions, structural distance ($\Delta S$), curvature ($\Phi$), and a regime convergence threshold (LCR). Experimental results show improved anticipatory capability compared to classical baselines. \end{abstract} \begin{IEEEkeywords} Structural Dynamics, Regime Detection, Semi-Markov, Non-Stationary Systems \end{IEEEkeywords} % ========================================================= \section{Introduction} Traditional models rely on value-based dynamics and fail to capture structural transitions. AURI-UCC shifts the paradigm to: \begin{itemize} \item persistence-driven relevance \item position-based structure \item convergence-based transitions \end{itemize} % ========================================================= \section{Model Definition} \subsection{State Representation} \begin{equation} E_i(t) = (A_i(t), I_i(t), D_i(t)) \end{equation} \begin{equation} A_i(t) = \mathbb{1}[x_i(t) > 0] \end{equation} \begin{equation} I_i(t) = x_i(t) - \mu_i(t) \end{equation} \begin{equation} D_i(t) = \text{sign}(x_i(t) - x_i(t-1)) \end{equation} % ========================================================= \subsection{Semi-Markov Persistence} \begin{equation} \tau_i(t) = \begin{cases} \tau_i(t-1) + 1 & \text{if } A_i(t)=A_i(t-1) \\ 1 & \text{otherwise} \end{cases} \end{equation} \begin{equation} P_i(t) = \min\left(\frac{\tau_i(t)}{W}, \theta_s\right) \end{equation} % ========================================================= \subsection{Mass} \begin{equation} M_i(t) = \left(\sum_{k=1}^{t} A_i(k)\right)(1 + P_i(t)) \end{equation} % ========================================================= \subsection{Structural Derivatives} \begin{equation} \Delta \pi_i(t) = \pi_i(t-1) - \pi_i(t) \end{equation} \begin{equation} \Delta^2 \pi_i(t) = \Delta \pi_i(t) - \Delta \pi_i(t-1) \end{equation} % ========================================================= \subsection{Structural Distance} \begin{equation} \Delta S_i(t) = w_A |\Delta A_i| + w_R |\Delta \pi_i| + w_M |\Delta M_i| + w_L |\Delta L_i| + w_C |\Phi_i| \end{equation} % ========================================================= \subsection{Curvature} \begin{equation} \Phi_i(t) = \alpha \overline{\Delta S_i} + \beta P_i(t) + \gamma L_i(t) + \delta Q_i(t) \end{equation} \begin{equation} Q_i(t) = |\Delta \pi_i| + \frac{1}{2} |\Delta^2 \pi_i| \end{equation} % ========================================================= \subsection{Structural Number} \begin{equation} Re_{GIQ}(i,t) = \frac{|\Delta \pi_i(t)| + \Delta S_i(t)}{P_i(t) + \epsilon} \end{equation} % ========================================================= \subsection{Decision Function} \begin{equation} Score_i(t) = 0.33 M_i(t) + 0.17 \Delta S_i(t) + 0.25 \Phi_i(t) + 0.25 P_i(t) \end{equation} % ========================================================= \section{Experimental Methodology} \subsection{Dataset} \begin{itemize} \item 50 sensors \item 5 regions \item 300 time steps \end{itemize} \subsection{Baselines} \begin{itemize} \item Random \item Frequency \item Markov \end{itemize} \subsection{Metrics} \begin{equation} F1 = 2 \cdot \frac{Precision \cdot Recall}{Precision + Recall} \end{equation} % ========================================================= \section{Results} \subsection{Quantitative Results} \begin{table}[h] \centering \begin{tabular}{lccc} \hline Model & Precision & Recall & F1 \\ \hline AURI-UCC & 0.43 & 1.00 & 0.60 \\ Markov & 0.47 & 1.00 & 0.64 \\ Frequency & 0.44 & 0.27 & 0.33 \\ Random & 0.09 & 0.20 & 0.12 \\ \hline \end{tabular} \caption{Performance Comparison} \end{table} \subsection{Analysis} Although Markov achieves slightly higher F1-score, it lacks anticipatory capability.
